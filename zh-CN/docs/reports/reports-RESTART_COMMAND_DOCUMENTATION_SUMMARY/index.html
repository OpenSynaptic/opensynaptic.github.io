Command Syntax (from README)​

# Basic: 10-second graceful shutdown (default)
os-node restart --graceful

# Custom timeout
os-node restart --graceful --timeout 5

# With custom server
os-node restart --graceful --timeout 15 --host 192.168.1.100 --port 9090
